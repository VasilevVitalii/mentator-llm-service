const entries = LIB.fs.readdirSync(args.dir)
const result = []
for (const entry of entries) {
    const fullPath = LIB.path.join(args.dir, entry)
    const stat = LIB.fs.statSync(fullPath)
    const ext = LIB.path.extname(entry)
    if (args.extension && ext.toLowerCase() !== args.extension.toLowerCase()) continue
    result.push({
        name: entry,
        extension: ext || null,
        sizeByte: stat.size,
        isDirectory: stat.isDirectory(),
        fullPath,
    })
}
return result
